cmake_minimum_required(VERSION 2.8)
PROJECT(automapd_sld)

FIND_PACKAGE(LibXml2) 

INCLUDE_DIRECTORIES("src" ${LIBXML2_INCLUDE_DIR} )

FILE(GLOB INTERFACE_HEADERS "interface/*.h")
FILE(GLOB INTERFACE_SOURCES "interface/*.cpp")
FILE(GLOB HEADERS "src/*.h")
FILE(GLOB SOURCES "src/*.cpp")

SOURCE_GROUP(Interface_Headers\\ FILES ${INTERFACE_HEADERS})
SOURCE_GROUP(Interface_Sources\\ FILES ${INTERFACE_SOURCES})

ADD_LIBRARY(automapd_sld ${HEADERS} ${SOURCES} ${INTERFACE_HEADERS} ${INTERFACE_SOURCES})
TARGET_LINK_LIBRARIES(automapd_sld ${LIBXML2_LIBRARIES} )
